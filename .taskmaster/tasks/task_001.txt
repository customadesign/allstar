# Description: Develop the core SignPreview component that accepts width, height, and unit as props and renders an SVG-based preview.
# Details:
Create a new React component with TypeScript:

```typescript
interface SignPreviewProps {
  width: number;
  height: number;
  unit: string;
}

const SignPreview: React.FC<SignPreviewProps> = ({ width, height, unit }) => {
  // Validate dimensions
  if (width <= 0 || height <= 0) {
    return <div className="sign-preview-error">Invalid dimensions</div>;
  }

  // Basic SVG structure with 220x220 viewBox
  return (
    <div className="sign-preview-container">
      <svg 
        viewBox="0 0 220 220" 
        xmlns="http://www.w3.org/2000/svg"
        className="sign-preview"
      >
        {/* Sign rectangle will be added in Task 2 */}
        {/* Labels and guides will be added in Task 3 */}
      </svg>
    </div>
  );
};

export default SignPreview;
```
